Parameters

//a method that has no parameters. use empty ( )
void PrintSomething()
{
}
//Parameters are just variables. Therefore, a parameter must have 2 things: a type and a name.
//here's a method that takes an int parameter called age. 
void PrintSomething(int age)
{
    Console.WriteLine($"Steve's age is {age}!");
}

//To call this method, an int constant or variable must be passed to the method
PrintSomething(12);//with a constant int

int stevesAge = 5;
PrintSomething(stevesAge);//with an int variable

Parameters: Pass by Value

//Pass by Value means the value of the parameter is COPIED to it when the method is called
void PrintSomething(int age) //age is a pass-by-value parameter
{
    Console.WriteLine($"Steve's age is {age}!");
}
int stevesAge = 5;
PrintSomething(stevesAge);//the value of stevesAge is COPIED to the age parameter in PrintSomething

//what is printed...
Steve's Age is 5!

Parameters: Pass by Reference

//Pass by Reference means the name of the parameter is an ALIAS 
//to the variable used when the method is called
//use the "ref" keyword to define the parameter as pass-by-reference
void GetAge(ref int age) //age is a pass-by-reference parameter. 
{
    age = 12; //changes to ref parameters modify the variable that is used when the method is called
}

//NOTE: the variable MUST be initialized before using it as an argument for a ref parameter
int stevesAge = 25; 
GetAge(ref stevesAge);//must use the "ref" keyword on reference parameters when calling the method.
Console.WriteLine($"Steve's age is {stevesAge}."); // prints "Steve's age is 12."

Parameters: Out parameters

//Out parameters are special Pass-by-Reference parameters.
//use the "out" keyword to define the parameter as out.
void GetAge(out int age) //age is out parameter. 
{
    age = 12; //the method MUST assign a value to every out parameter
}

int stevesAge; //do NOT need to initialize the variable before using it as an argument for out parameters
GetAge(out stevesAge);//must use the "out" keyword on out parameters when calling the method
Console.WriteLine($"Steve's age is {stevesAge}."); // prints "Steve's age is 12."

Parameters: Optional parameters

//optional parameters do not need to be passed when calling the method.
//ALL optional parameters must appear at the end of the parameter list
//the name parameter is REQUIRED. The age parameter is optional.
void PrintSomething(string name, int age = 12) 
{
    Console.WriteLine($"{name}'s age is {age}!");
}

PrintSomething("Steve", 7);  //prints "Steve's age is 7!"
PrintSomething("Steve");     //prints "Steve's age is 12!"   the default value of 12 is used.
