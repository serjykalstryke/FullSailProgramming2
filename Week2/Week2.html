Sorting and Searching

Comparing Items:

string s1 = "Batman", s2 = "Aquaman";
int compResult = s1.CompareTo(s2);
if(compResult < 0) 
    Console.WriteLine($"{s1} is LESS THAN {s2}");
else if(compResult == 0) 
    Console.WriteLine($"{s1} is EQUAL TO {s2}");
else if(compResult > 0) 
    Console.WriteLine($"{s1} is GREATER THAN {s2}");

Swapping Items:

int[] nums = new int[] {1,12,23,45,34};
int index1 = 3; //3 is the index of 45 in the array
int index2 = 4; //4 is the index of 34 in the array
int temp = nums[index1]; //save the value at index1 to a temporary variable
nums[index1] = nums[index2]; //overwrite the value stored at index1
nums[index2] = temp;    //overwrite the value stored at index2
int[] nums = new int[] {1,12,23,45,34};
int index1 = 3; //3 is the index of 45 in the array
int index2 = 4; //4 is the index of 34 in the array
(nums[index2], nums[index1]) = (nums[index1], nums[index2]);

Recursion

int i = 0;
RecursiveMethod(i);

static void RecursiveMethod(int i)
{
    if(i < 100) //the exit condition
    {
        Console.WriteLine(i);
        RecursiveMethod(i+1);//the recursive call
    }
}

Dictionaries

Create a Dictionary:

Dictionary<string,StudentRecord> studentInfo = new Dictionary<string,StudentRecord>();

Add items in the initializer:

Dictionary<string,float> menu = new Dictionary<string,float>()  //don't put a ; here
{
    //{ key, value }
    {"Hamburger", 1.99F},
    {"Cheesy Hamburger", 2.99F}
};  //put the ; here

Add a key and value using the Add method:

menu.Add("French Fries", 0.99F);
menu.Add("Soda", 0.99F);

Add a key and value using [ ]:

menu["Chicken Nuggets"] = 3.99F;

Loop over the Dictionary

foreach(KeyValuePair<string,float> menuItem in menu)
{
    //the key will be the name of the menu item (EX: Hamburger)
    //the value will be the price of the menu item (EX: 1.99)
    Console.WriteLine($"{menuItem.Key}: {menuItem.Value:C2}");
}

How to remove a key and its value

string itemToRemove = "Chicken Nuggets";
bool wasRemoved = menu.Remove(itemToRemove);
if(wasRemoved)
    Console.WriteLine($"{itemToRemove} was removed from the menu.");
else
    Console.WriteLine($"{itemToRemove} was not found.");

ContainsKey

Dictionary<string,float> menu = new Dictionary<string,float>()  
{
    {"Hamburger", 1.99F},
    {"Cheesy Hamburger", 2.99F}
};  
bool isFound = menu.ContainsKey("Iced Tea");
Console.WriteLine(isFound);

TryGetValue

Dictionary<string,float> menu = new Dictionary<string,float>()  
{
    {"Hamburger", 1.99F},
    {"Cheesy Hamburger", 2.99F}
};  
string keyToFind = "Cheesy Hamburger";
bool isFound = menu.TryGetValue(keyToFind, out float price);
if(isFound)
    Console.WriteLine($"{keyToFind} costs {price:C2}");

Updating a Value

Dictionary<string,float> menu = new Dictionary<string,float>()  
{
    {"Hamburger", 1.99F},
    {"Cheesy Hamburger", 2.99F}
};  
string keyToFind = "Cheesy Hamburger";
float price = menu[keyToFind]; //NOTE: if you're not sure if the key is in the dictionary, use ContainsKey to check first otherwise it'll throw an exception
menu[keyToFind] = price + 1F;
Console.WriteLine($"{keyToFind} now costs {menu[keyToFind]:C2}");
